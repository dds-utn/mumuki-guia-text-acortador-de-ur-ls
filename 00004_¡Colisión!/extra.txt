public class MD5 {
  public static String generar(String entrada) {
    Map<String, String> urls = new HashMap<>();
    urls.put("http://misitio.com/foo", "AAA");
    urls.put("http://misitio.com/bar", "BBB");
    urls.put("http://misitio.com/baz", "CCC");
    urls.put("http://misitio.com/foobar", "AAA");
    urls.put("http://misitio.com/foobar2", "AAA");
    return urls.getOrDefault(entrada, entrada);
  }
}
